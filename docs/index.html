FRAGAttribute ::= Subject Operator "have" ("a" | "an") Measurable "of" Expression Units.
FRAGRequirement ::= ("Otherwise," | (Preconditions* Trigger?) Subject Operator Action Constraints "."
Preconditions ::= "While" (Measurable Comparitor Expression Units ("&" | ",")+
Trigger ::= ("when" | "When") Measurable "becomes" Comparitor? Expression Units ","
Subject ::= "the" SystemName
Operator ::= "must" | "shall" | "should" | "may" | "can"
Action ::= ("produce" | "set" | "actuate") Measurable ("as" | "to") Expression Units

Constraints ::= "with" (("a" | "an") Measurable "of" Comparitor Expression Units ("," | "and"))+
